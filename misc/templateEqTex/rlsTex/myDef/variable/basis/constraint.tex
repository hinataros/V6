\NewDocumentCommand\mbbBn{m o o o}
{
  \IfValueTF{#2}
  {
    \expTwoTThrT{\mbb B_{#1}^{#2}}%
    {\bbmat{\mbbB_{{#1}_f}^{#2}\\ \mbbB_{#1_m}^{#2}}}%
    {\bbmat{(\mbbB_{#1_f}^{#2})^T & (\mbbB_{#1_m}^{#2})^T} }%
    {error}%
    {error}%
    [#3][#4]
  }
  {
    \expTwoTThrT{\mbb B_{#1}}%

    {\bbmat{\mbbB_{#1_f}^T&\mbbB_{#1_m}^T}}%
    {error}%
    {error}%
    [#3][#4]
  }
}
